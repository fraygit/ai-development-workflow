### `webhook-bridge/index.js`

```js
const http = require('http')
const crypto = require('crypto')

const {
  JIRA_WEBHOOK_SECRET,
  GITHUB_TOKEN,
  GITHUB_REPO,              // e.g. "fraygit/ai-development-workflow-poc"
  SYSTEM_USER_ACCOUNT_ID,   // Jira accountId of the bot user
  PORT = '3000'
} = process.env

if (!GITHUB_TOKEN || !GITHUB_REPO || !SYSTEM_USER_ACCOUNT_ID) {
  console.error('Missing required env vars: GITHUB_TOKEN, GITHUB_REPO, SYSTEM_USER_ACCOUNT_ID')
  process.exit(1)
}

const server = http.createServer((req, res) => {
  if (req.method !== 'POST' || req.url !== '/webhook/jira') {
    res.writeHead(404)
    return res.end('Not found')
  }

  let body = ''
  req.on('data', chunk => { body += chunk })
  req.on('end', async () => {
    try {
      // Verify HMAC signature if a secret is configured
      const signature = req.headers['x-hub-signature-256']
      if (JIRA_WEBHOOK_SECRET) {
        if (!signature) {
          res.writeHead(401)
          return res.end('Missing signature')
        }
        const expected = 'sha256=' + crypto
          .createHmac('sha256', JIRA_WEBHOOK_SECRET)
          .update(body)
          .digest('hex')
        if (!crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(expected))) {
          res.writeHead(401)
          return res.end('Invalid signature')
        }
      }

      const payload = JSON.parse(body)

      // Only act on issue_updated events where the assignee changed to our system user
      if (payload.webhookEvent !== 'jira:issue_updated') {
        res.writeHead(200)
        return res.end('OK')
      }

      const assignee = payload.issue?.fields?.assignee
      if (!assignee || assignee.accountId !== SYSTEM_USER_ACCOUNT_ID) {
        res.writeHead(200)
        return res.end('OK')
      }

      const ticketId = payload.issue.key
      console.log(`Dispatching GitHub Actions for ticket: ${ticketId}`)

      const response = await fetch(
        `https://api.github.com/repos/${GITHUB_REPO}/dispatches`,
        {
          method: 'POST',
          headers: {
            'Accept': 'application/vnd.github+json',
            'Authorization': `Bearer ${GITHUB_TOKEN}`,
            'X-GitHub-Api-Version': '2022-11-28',
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            event_type: 'jira-ticket-assigned',
            client_payload: { ticket_id: ticketId }
          })
        }
      )

      if (!response.ok) {
        const text = await response.text()
        console.error(`GitHub dispatch failed: ${response.status} ${text}`)
        res.writeHead(502)
        return res.end('GitHub dispatch failed')
      }

      console.log(`Dispatched: ${ticketId}`)
      res.writeHead(200)
      res.end('OK')

    } catch (err) {
      console.error('Error processing webhook:', err)
      res.writeHead(500)
      res.end('Internal error')
    }
  })
})

server.listen(parseInt(PORT), () => {
  console.log(`Webhook bridge listening on port ${PORT}`)
})
```

## Acceptance Criteria

**Code**
- [ ] Server listens on `POST /webhook/jira` only; all other paths return `404`.
- [ ] HMAC-SHA256 signature verified via `crypto.timingSafeEqual` when `JIRA_WEBHOOK_SECRET` is set. Missing or invalid signature → `401`.
- [ ] Event ignored (returns `200 OK`) if `webhookEvent` is not `jira:issue_updated` or the assignee is not the configured system user.
- [ ] Dispatches `repository_dispatch` with `event_type: "jira-ticket-assigned"` and `client_payload: { ticket_id }` to the configured repo.
- [ ] Missing required env vars at startup → logs error and exits immediately (no silent partial startup).

**Deployment**
- [ ] Deployed to Railway / Render / Fly.io (free tier) with a public HTTPS URL.
- [ ] Required env vars configured in the hosting platform's dashboard:
  - `GITHUB_TOKEN` — fine-grained PAT with `Actions: write` and `Contents: read` on the POC repo
  - `GITHUB_REPO` — `fraygit/ai-development-workflow-poc`
  - `SYSTEM_USER_ACCOUNT_ID` — Jira accountId of the bot user
  - `JIRA_WEBHOOK_SECRET` — the secret configured in the Jira webhook (optional but recommended)
- [ ] Jira webhook registered at `https://<bridge-url>/webhook/jira` for the `Issue Updated` event.

**End-to-end test**
- [ ] Assign a Jira ticket to the system user → GitHub Actions workflow triggers within 30 seconds → PR created.

## Notes

- **GitHub Token scope:** use a fine-grained PAT (not a classic PAT). Required permissions: `Actions: write` (to dispatch), `Contents: read`. Repository access: POC repo only.
- **Jira webhook limitations:** Jira Cloud supports registering webhooks via `Project settings → Automation → Webhooks` or via the Jira REST API. The webhook fires for all issues in the project — the bridge filters by assignee accountId server-side.
- **No `repository_dispatch` for the analyse workflow yet:** the bridge currently only handles `jira-ticket-assigned`. When Story 2 is wired, add a second system user account ID and dispatch `jira-ticket-assigned-for-analysis` for that user's assignments.
- **Deployment recommendation:** Railway is the simplest free option for a Node.js server. Connect the POC repo, select the `webhook-bridge/` directory as the root, set the start command to `node index.js`.
